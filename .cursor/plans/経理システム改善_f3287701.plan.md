# 経理システム改善プラン

## 現状分析

既存の基盤は思ったより整っている:

- 仕訳テーブル (`accounting_journal_entries`, `accounting_journal_lines`) は既に存在
- 勘定科目コード (`account_code`) の概念あり

## Phase 1: 監査対応強化（内部統制の土台）

### 1.1 承認権限の階層化

**現状の問題:**

```typescript
// accounting.ts L205-228 - 誰でも承認できる
.eq("reviewer_id", req.userId!)  // 割り当てられた人のみだが、金額制限なし
```

**改善:**

- `profiles` テーブルに `approval_limit` カラム追加
- 金額による承認者階層（例: 5万円以下=一般、5-30万円=リーダー、30万円超=マネージャー）
```sql
-- profiles テーブルに追加
ALTER TABLE profiles ADD COLUMN IF NOT EXISTS approval_limit numeric DEFAULT 50000;
ALTER TABLE profiles ADD COLUMN IF NOT EXISTS role text DEFAULT 'member' 
  CHECK (role IN ('member', 'leader', 'manager', 'admin'));
```


**承認者割当ロジック変更:**

```typescript
// rpc_assign_random_reviewer を改修
// 金額に応じた承認権限を持つユーザーを選択
SELECT id INTO v_reviewer
FROM profiles
WHERE id <> v_created_by
  AND approval_limit >= v_amount
ORDER BY approval_limit ASC, random()
LIMIT 1;
```

### 1.2 変更履歴テーブル追加

```sql
CREATE TABLE IF NOT EXISTS public.accounting_audit_log (
  id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  table_name text NOT NULL,
  record_id uuid NOT NULL,
  action text NOT NULL CHECK (action IN ('INSERT', 'UPDATE', 'DELETE')),
  old_values jsonb,
  new_values jsonb,
  changed_by uuid NOT NULL REFERENCES auth.users(id),
  changed_at timestamptz DEFAULT now(),
  ip_address text,
  user_agent text
);

CREATE INDEX audit_log_record_idx ON accounting_audit_log (table_name, record_id);
CREATE INDEX audit_log_changed_at_idx ON accounting_audit_log (changed_at DESC);
```

**トリガーで自動記録:**

```sql
CREATE OR REPLACE FUNCTION audit_trigger()
RETURNS trigger AS $$
BEGIN
  INSERT INTO accounting_audit_log (table_name, record_id, action, old_values, new_values, changed_by)
  VALUES (
    TG_TABLE_NAME,
    COALESCE(NEW.id, OLD.id),
    TG_OP,
    CASE WHEN TG_OP != 'INSERT' THEN row_to_json(OLD) END,
    CASE WHEN TG_OP != 'DELETE' THEN row_to_json(NEW) END,
    COALESCE(NEW.created_by, OLD.created_by, auth.uid())
  );
  RETURN NEW;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;
```

interface JournalLine {
  accountCode: string;
  accountName: string;
  debit: number;
  credit: number;
}

function JournalPreview({ lines }: { lines: JournalLine[] }) {
  const totalDebit = lines.reduce((sum, l) => sum + l.debit, 0);
  const totalCredit = lines.reduce((sum, l) => sum + l.credit, 0);
  const isBalanced = totalDebit === totalCredit;

  return (
    <div className={styles.journalPreview}>
      <div className={styles.header}>
        <span>仕訳プレビュー</span>
        {isBalanced ? <CheckCircle /> : <AlertTriangle />}
      </div>
      <table>
        <thead>
          <tr><th>勘定科目</th><th>借方</th><th>貸方</th></tr>
        </thead>
        <tbody>
          {lines.map((line, i) => (
            <tr key={i}>
              <td>{line.accountCode} {line.accountName}</td>
              <td>{line.debit > 0 ? `¥${line.debit.toLocaleString()}` : ''}</td>
              <td>{line.credit > 0 ? `¥${line.credit.toLocaleString()}` : ''}</td>
            </tr>
          ))}
        </tbody>
        <tfoot>
          <tr>
            <td>合計</td>
            <td>¥{totalDebit.toLocaleString()}</td>
            <td>¥{totalCredit.toLocaleString()}</td>
          </tr>
        </tfoot>
      </table>
    </div>
  );
}

## Phase 4: バッチ承認

### 4.1 バックエンドAPI追加

[server/src/routes/accounting.ts](server/src/routes/accounting.ts) に追加:

```typescript
// 複数経費の一括承認
router.post("/expenses/batch-review", async (req: AuthenticatedRequest, res: Response) => {
  const { ids, action, comment } = req.body;
  
  if (!Array.isArray(ids) || ids.length === 0) {
    return res.status(400).json({ error: "ids array is required" });
  }
  
  if (ids.length > 50) {
    return res.status(400).json({ error: "Maximum 50 items per batch" });
  }

  const results = { success: [], failed: [] };
  
  for (const id of ids) {
    try {
      // 自己承認チェック
      const { data: tx } = await supabaseAdmin
        .from("accounting_transactions")
        .select("created_by, amount_total")
        .eq("id", id)
        .single();
      
      if (tx?.created_by === req.userId) {
        results.failed.push({ id, error: "自己承認不可" });
        continue;
      }
      
      // 承認権限チェック
      const { data: profile } = await supabaseAdmin
        .from("profiles")
        .select("approval_limit")
        .eq("id", req.userId)
        .single();
      
      if ((profile?.approval_limit || 0) < (tx?.amount_total || 0)) {
        results.failed.push({ id, error: "承認権限不足" });
        continue;
      }
      
      // 承認実行
      await supabaseAdmin
        .from("accounting_transactions")
        .update({
          review_status: action === "approve" ? "approved" : "rejected",
          review_comment: comment,
          reviewed_at: new Date().toISOString(),
          status: action === "approve" ? "posted" : "draft",
        })
        .eq("id", id)
        .eq("reviewer_id", req.userId);
      
      results.success.push(id);
    } catch (err: any) {
      results.failed.push({ id, error: err.message });
    }
  }
  
  res.json(results);
});
```
